Examples  

Size


<input type="checkbox" checked data-size="large">
<input type="checkbox" checked data-size="normal">
<input type="checkbox" checked data-size="small">
<input id="switch-size" type="checkbox" checked data-size="mini">
$('#switch-size').MoUISwitch('size','mini');
$('#switch-size').MoUISwitch('size','small');
$('#switch-size').MoUISwitch('size','normal');
$('#switch-size').MoUISwitch('size','large');
alert($('#switch-size').MoUISwitch('size'));

State


<input id="switch-state" type="checkbox" checked>
$('#switch-state').MoUISwitch('toggleState');
$('#switch-state').MoUISwitch('state',true);
$('#switch-state').MoUISwitch('state',false);
alert($('#switch-state').MoUISwitch('state'));
